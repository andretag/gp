55  class Plot1D : public MatPlot {
56  public:
57  ~Plot1D() {}
58  Plot1D(const GaussianProcess* const gp, double xmin, double xmax,
59  double ymin, double ymax, size_t num_points = 100,
60  const std::string& title = "", const std::string& xlabel = "",
61  const std::string& ylabel = "")
62  : gp_(gp),
63  xmin_(xmin), xmax_(xmax),
64  ymin_(ymin), ymax_(ymax),
65  num_points_(num_points),
66  title_(title),
67  xlabel_(xlabel),
68  ylabel_(ylabel),
69  MatPlot() {
70  CHECK_NOTNULL(gp);
71  CHECK_EQ(gp_->Dimension(), 1);
72  }
73 
74  // Implement pure virtual display function.
75  void DISPLAY() {
76  // Sample the x axis.
77  const std::vector<double> x = linspace(xmin_, xmax_, num_points_);
78 
79  // Three functions to compute: mean, upper bound, lower bound.
80  std::vector<double> mu(num_points_);
81  std::vector<double> upper(num_points_);
82  std::vector<double> lower(num_points_);
83 
84  double mean, variance;
85  VectorXd test_point(1);
86  for (size_t ii = 0; ii < num_points_; ii++) {
87  // Compute the GP mean and variance here.
88  test_point(0) = x[ii];
89  gp_->Evaluate(test_point, mean, variance);
90 
91  // Populate mean, upper, lower.
92  mu[ii] = mean;
93  upper[ii] = mean + 2.0 * std::sqrt(variance);
94  lower[ii] = mean - 2.0 * std::sqrt(variance);
95  }
96 
97  // Also unpack the GP training points.
98  const ConstPointSet points = gp_->ImmutablePoints();
99  const VectorXd targets = gp_->ImmutableTargets();
100  CHECK_EQ(points->size(), targets.size());
101 
102  std::vector<double> training_x(points->size());
103  std::vector<double> training_y(points->size());
104  for (size_t ii = 0; ii < points->size(); ii++) {
105  training_x[ii] = points->at(ii)(0);
106  training_y[ii] = targets(ii);
107  }
108 
109  // Plot.
110  plot(x, mu); set(2); set("g");
111  plot(x, upper); set(2); set("r"); set(":");
112  plot(x, lower); set(2); set("r"); set(":");
113  plot(training_x, training_y); set(3); set("p"); set("*");
114 
115  axis(xmin_, xmax_, ymin_, ymax_);
116 
117  title(title_);
118  xlabel(xlabel_);
119  ylabel(ylabel_);
120  }
121 
122  private:
123  // Gaussian process pointer.
124  const GaussianProcess* const gp_;
125 
126  // Plot params.
127  const double xmin_, xmax_, ymin_, ymax_;
128  const size_t num_points_;
129 
130  // Title and labels.
131  const std::string title_, xlabel_, ylabel_;
132  }; //\ class Plot1D
